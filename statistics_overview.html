import math
def getsd(ls):
    summ=0
    x_mean=sum(ls)/len(ls)
    print("Mean :", x_mean)
    for l in ls:
        summ=summ+((l-x_mean)**2)
    print("Standard Deviation :",math.sqrt(summ/len(ls)))

def getcr(x, y):
    A=0
    B=0
    C=0
    x_mean=sum(x)/len(x)
    y_mean=sum(y)/len(y)
    for r in range(len(x)):#0 1 2 3 4
        A=A + (x[r]-x_mean)*(y[r]-y_mean)
        B=B + (x[r]-x_mean)**2
        C=C + (y[r]-y_mean)**2
    
    print("CR :", A/math.sqrt(B*C))
